\begin{align*}
\scaleint{6ex} \big[ Y_{lm} \big]^{*} \, L_{-} \, \big( L_{+} \, Y_{lm} \big) \dd{\Omega} = \scaleint{6ex} \big( L_{+} \, Y_{lm} \big)^{*} \, \big( L_{+} \, Y_{lm} \big) \dd{\Omega}
\end{align*}
