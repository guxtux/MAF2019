\documentclass[border=5mm]{standalone}
\usepackage{bm}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\begin{document}
    \tdplotsetmaincoords{60}{110}
%
    \pgfmathsetmacro{\rvec}{.8}
    \pgfmathsetmacro{\thetavec}{30}
    \pgfmathsetmacro{\phivec}{60}
%
\begin{tikzpicture}[scale=5,tdplot_main_coords]
    \coordinate (O) at (0,0,0);

    \draw[thick,->] (0,0,0) -- (1,0,0) node[anchor=north east]{$x$};

    \draw[thick,->] (0,0,0) -- (0,1,0) node[anchor=north west]{$y$};

    \draw[thick,->] (0,0,0) -- (0,0,1) node[anchor=south]{$z$};
    
    \tdplotsetcoord{P}{\rvec}{\thetavec}{\phivec}

    \draw[-stealth] (O) -- (P) node [midway, below] {$\rho$};
    
    \draw (O) -- (0.4, 0.7, 0);
    
    \tdplotdrawarc[-stealth]{(O)}{0.2}{0}{\phivec}{anchor=north}{$\phi$}

    \tdplotsetthetaplanecoords{\phivec}

    \tdplotdrawarc[tdplot_rotated_coords,-stealth]{(0,0,0)}{0.5}{0}%
        {\thetavec}{anchor=south west}{$\theta$}
    
    \draw[tdplot_rotated_coords] (\rvec, 0, 0) arc (0:90:\rvec);
    
    \draw (\rvec, 0, 0) arc (0:90:\rvec);
    
\end{tikzpicture}
\end{document}
